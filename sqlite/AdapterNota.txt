package com.ifsc.sqlite;

import android.content.Context;
import android.view.LayoutInflater;
import android.view.View;
import android.view.ViewGroup;
import android.widget.ArrayAdapter;
import android.widget.TextView;

import androidx.annotation.NonNull;
import androidx.annotation.Nullable;

import java.util.List;

// AdapterNota é um ArrayAdapter personalizado que exibe objetos da classe Nota
public class AdapterNota extends ArrayAdapter<Nota> {

    // Armazena o ID do layout que será inflado para cada item da lista
    int myResource;

    // Construtor do adapter
    public AdapterNota(@NonNull Context context, int resource, @NonNull List<Nota> objects) {
        // Chama o construtor da superclasse (ArrayAdapter)
        super(context, resource, objects);

        // Salva o ID do recurso de layout passado como parâmetro
        myResource = resource;
    }

    // Método chamado automaticamente para montar cada item da lista visualmente
    @NonNull
    @Override
    public View getView(int position, @Nullable View convertView, @NonNull ViewGroup parent) {

        // Cria um "inflador" de layouts baseado no contexto atual
        LayoutInflater layoutInflater = LayoutInflater.from(getContext());

        // Infla (cria) a visualização do layout do item da lista
        View v = layoutInflater.inflate(myResource, parent, false);

        // Obtém o objeto Nota referente à posição atual da lista
        Nota nota = getItem(position);

        // Referencia o TextView dentro do layout (usando o ID padrão do Android)
        TextView tv = v.findViewById(android.R.id.text1);

        // Define o texto do TextView como sendo o conteúdo da nota
        tv.setText(nota.getTexto());

        // Retorna a visualização pronta do item
        return v;
    }
}
